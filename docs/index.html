// Get a desaturated base color
function getBaseColor() {
  let baseColor = chroma.random()
  while (chroma(baseColor).hsl()[1] > 1 / 3) {
    baseColor = chroma(baseColor).desaturate(1 / 8).hex()
  }
  return baseColor
}
